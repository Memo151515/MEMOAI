from flask import Flask, request, jsonify import requests import os app = Flask(__name__) # NOT: Buraya OpenRouter veya benzeri bir API Key eklemelisin. # Vercel Settings -> Environment Variables kısmına "API_KEY" adıyla ekle. API_KEY = os.getenv("API_KEY") @app.route('/api/chat', methods=['POST']) def chat(): data = request.json user_message = data.get("prompt") # MEMO AI Kimlik Bilgisi system_prompt = "Sen MEMO AI'sın. Kurucun Mehmet Can YILDIRIM'dır. Otonom ve zekisin." try: # İnternet tabanlı model (Ücretsiz bir model seçebilirsin) response = requests.post( url="https://openrouter.ai/api/v1/chat/completions", headers={"Authorization": f"Bearer {API_KEY}"}, json={ "model": "google/gemini-pro-1.5-exp", # Veya istediğin bir model "messages": [ {"role": "system", "content": system_prompt}, {"role": "user", "content": user_message} ] } ) return jsonify(response.json()) except Exception as e: return jsonify({"error": str(e)}), 500
